# conditional statements
temperature = 25
if temperature > 30 :
      print("It's a hot day")
elif temperature > 20 :
      print("It's a nice day")
else :
      print("It's a cold day")

# Loops
for i in range(10):
      print(i)
for i in range(4,8): # this will print numbers from 4 to 7
      print(i)
for x in "Python": # this will print each character in the string "Python"
      print(x)
while temperature > 20:
      print("It's a nice day")
      temperature -= 1 # this will decrease the temperature by 2 in each iteration of the loop
